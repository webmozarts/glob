26: class GlobIterator extends GlobFilterIterator
27: {
28:     /**
29:      * Creates a new iterator.
30:      *
31:      * @param string $glob  The glob pattern.
32:      * @param int    $flags A bitwise combination of the flag constants in
33:      *                      {@link Glob}.
34:      */
35:     public function __construct($glob, $flags = 0)
36:     {
37:         $basePath = Glob::getBasePath($glob);
38: 
39:         if (file_exists($glob) && false === strpos($glob, '*')) {
40:             // If the glob is a file path, return that path
41:             $innerIterator = new ArrayIterator(array($glob));
42:         } elseif (is_dir($basePath)) {
43:             // Otherwise scan the glob's base directory for matches
44:             $innerIterator = new RecursiveIteratorIterator(
45:                 new RecursiveDirectoryIterator($basePath),
46:                 RecursiveIteratorIterator::SELF_FIRST
47:             );
48:         } else {
49:             // If the glob's base directory does not exist, return nothing
50:             $innerIterator = new EmptyIterator();
51:         }
52: 
53:         parent::__construct($glob, $innerIterator, self::FILTER_VALUE, $flags);
